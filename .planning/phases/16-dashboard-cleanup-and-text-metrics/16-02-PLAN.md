    The TextMetrics weekly summary needs intake records by date range. Rather than importing `getRecordsByDateRange` from intake-service directly into a component (which violates the established service -> hook -> component pattern in this codebase), add a new hook to use-intake-queries.ts:

    ```typescript
    /**
     * Hook to get intake records within a date range, optionally filtered by type.
     * Uses useLiveQuery for reactive updates. Re-runs every 60s via tick dep.
     */
    export function useIntakeRecordsByDateRange(
      startTime: number,
      endTime: number,
      type?: "water" | "salt"
    ) {
      const [tick, setTick] = useState(0);
      useEffect(() => {
        const interval = setInterval(() => setTick(t => t + 1), 60_000);
        return () => clearInterval(interval);
      }, []);

      return useLiveQuery(
        () => getRecordsByDateRange(startTime, endTime, type),
        [startTime, endTime, type, tick],
        []
      );
    }
    ```

    Add the `getRecordsByDateRange` import from `@/lib/intake-service` at the top of use-intake-queries.ts (alongside the existing imports from that module). This is fine -- hooks files ARE the correct place to import service functions.
