    ```typescript
    #!/usr/bin/env tsx
    /**
     * Reset the Neon Postgres public schema to a clean state.
     *
     * Drops every app table (both the Drizzle-owned names and the legacy push table
     * name `push_dose_schedules`) plus the drizzle-kit migration tracker. Uses
     * `DROP TABLE IF EXISTS ... CASCADE`, so it's idempotent and handles inner FK
     * dependencies automatically.
     *
     * PRESERVES: the entire `neon_auth` schema. This script never touches tables
     * outside the default `public` schema.
     *
     * Usage:
     *   DATABASE_URL=postgres://... pnpm db:reset
     *
     * Safety:
     *   - No interactive confirmation â€” only run against a branch you know is safe to wipe.
     *   - NEVER run against production (production is main branch â€” use a feature branch).
     *   - CI runs against ephemeral branches that are deleted afterward anyway.
     */
    import { neon } from "@neondatabase/serverless";
    import "dotenv/config";

    const TABLES: readonly string[] = [
      // 16 Dexie-mirrored app tables (new Drizzle names)
      "inventory_transactions",     // drop child before parent (CASCADE would cover it, but cleaner)
      "dose_logs",
      "daily_notes",
      "substance_records",
      "intake_records",
      "eating_records",
      "weight_records",
      "blood_pressure_records",
      "urination_records",
      "defecation_records",
      "audit_logs",
      "phase_schedules",
      "inventory_items",
      "medication_phases",
      "titration_plans",
      "prescriptions",
      // 4 push tables (new names)
      "push_sent_log",
      "push_schedules",
      "push_subscriptions",
      "push_settings",
      // Legacy push name (dropped if still present from pre-Phase-42 state)
      "push_dose_schedules",
      // drizzle-kit migration tracker (drop so next migrate re-applies)
      "__drizzle_migrations",
    ] as const;

    async function main(): Promise<void> {
      const url = process.env.DATABASE_URL;
      if (!url) {
        throw new Error("DATABASE_URL not set â€” export it or add it to .env.local");
      }

      // NEVER allow this script to run against production.
      // Simple heuristic: reject if the connection string mentions `prod`.
      // Operators can override by setting ALLOW_PROD_RESET=1 (intentional footgun).
      if (/prod/i.test(url) && process.env.ALLOW_PROD_RESET !== "1") {
        throw new Error(
          "DATABASE_URL appears to point at production. Refusing to wipe. " +
          "Set ALLOW_PROD_RESET=1 to override (you probably shouldn't)."
        );
      }

      const sql = neon(url);

      console.log(`Resetting Neon public schema (${TABLES.length} tables)...`);
      for (const table of TABLES) {
        console.log(`  DROP TABLE IF EXISTS ${table} CASCADE`);
        // neon() exposes a tagged-template SQL method; use sql.query() for dynamic
        // identifiers since tagged-template-only mode cannot splice identifiers.
        await sql.query(`DROP TABLE IF EXISTS ${table} CASCADE`);
      }

      console.log("");
      console.log("âś“ Neon public schema reset. neon_auth schema preserved.");
      console.log("  Next step: pnpm db:migrate");
    }

    main().catch((err) => {
      console.error("reset-neon-db failed:", err);
      process.exit(1);
    });
    ```

    **Tagged-template vs query() note:** The `@neondatabase/serverless` package exposes `neon(url)` which returns a client. The tagged-template form (`` sql`SELECT 1` ``) handles parameterized queries safely but cannot splice identifiers (table names). For DDL with dynamic identifiers, use `sql.query(textSQL)` (positional-param form) or `sql.unsafe(textSQL)` if the current version exposes that. If the current version of `@neondatabase/serverless` (^1.0.2) does not expose `sql.query`, adapt to whatever DDL escape hatch IS exposed â€” consult `node_modules/@neondatabase/serverless/index.d.ts` for the signature. Identifier injection risk is nil here because `TABLES` is a compile-time constant array of string literals.

    **Why drop order matters even with CASCADE:** CASCADE handles it automatically, but listing children before parents gives clearer operator output (line-by-line: "dropping inventory_transactions... dropping dose_logs...") and matches reverse topological order documented in Â§5 of 42-RESEARCH.md.

    **Why exclude neon_auth:** This script's `TABLES` array contains only bare names (no schema prefix). All `DROP TABLE` calls run against the default `public` schema. Even if an operator accidentally adds a table name that also exists under `neon_auth`, Postgres will only drop from `public` because unqualified names resolve via `search_path` which defaults to `public` first. But to make this absolutely clear, add a runtime assertion at the top of `main()`:
    ```typescript
    if (TABLES.some((t) => t.includes("."))) {
      throw new Error("TABLES must contain unqualified names only â€” schema-qualified names are forbidden to prevent accidental neon_auth drops");
    }
    ```
