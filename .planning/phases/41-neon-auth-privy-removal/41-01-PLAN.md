    1. Rewrite `src/lib/auth-middleware.ts` (full replacement — delete the old Bearer-token logic):
       ```typescript
       import { NextRequest, NextResponse } from "next/server";
       import { auth } from "./neon-auth";

       export interface VerificationResult {
         success: boolean;
         userId?: string;
         email?: string;
         error?: string;
       }

       export interface AuthenticatedRequest {
         request: NextRequest;
         auth: VerificationResult;
       }

       type AuthenticatedHandler = (
         ctx: AuthenticatedRequest
       ) => Promise<NextResponse> | NextResponse;

       function getAllowedEmails(): string[] {
         return (process.env.ALLOWED_EMAILS || "")
           .split(",")
           .map((e) => e.trim().toLowerCase())
           .filter(Boolean);
       }

       /**
        * Higher-order function that wraps an API route handler with Neon Auth session verification
        * + ALLOWED_EMAILS whitelist enforcement.
        *
        * Session is read from the signed cookie set by Neon Auth. No Bearer token plumbing.
        *
        * Usage (unchanged from v1):
        * ```ts
        * export const POST = withAuth(async ({ request, auth }) => {
        *   // auth.userId! is guaranteed on success
        *   return NextResponse.json({ ok: true });
        * });
        * ```
        */
       export function withAuth(handler: AuthenticatedHandler) {
         return async (request: NextRequest): Promise<NextResponse> => {
           const session = await auth.getSession(request);

           if (!session?.user) {
             return NextResponse.json(
               { error: "No active session", requiresAuth: true },
               { status: 401 }
             );
           }

           const userEmail = session.user.email?.toLowerCase();
           const userId = session.user.id;
           const allowedEmails = getAllowedEmails();

           // Whitelist check — empty whitelist allows any authenticated email
           // (preserves current Privy behavior from privy-server.ts lines 118-127)
           if (allowedEmails.length > 0 && (!userEmail || !allowedEmails.includes(userEmail))) {
             return NextResponse.json(
               {
                 error: "Your account is not authorized to use this app",
                 requiresAuth: true,
               },
               { status: 401 }
             );
           }

           return handler({
             request,
             auth: {
               success: true,
               userId,
               ...(userEmail !== undefined && { email: userEmail }),
             },
           });
         };
       }
       ```
       Note: this keeps the `VerificationResult` interface shape identical so route handlers that destructure `auth.userId!` and `auth.email` still work (D-06 goal: zero changes to the 9 existing routes). The file no longer imports `privy-server.ts` — that file will be deleted in plan 41-04.

       If `auth.getSession()` in the actual Neon Auth API requires a different call shape (e.g., a Headers object instead of a Request), consult Context7 docs and adapt. The test file mocks this, so adjust the mock accordingly.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → /api/auth/[...path] | Untrusted client traffic hits Neon Auth handler — must validate all inputs via Better Auth internals |
| client cookies → withAuth | Signed session cookie crosses into server; signature verified via NEON_AUTH_COOKIE_SECRET |
| ALLOWED_EMAILS env → whitelist | Server-controlled config gate, not attacker-controlled |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-41-01 | Spoofing | `auth.getSession()` in withAuth | mitigate | Neon Auth uses signed cookies via NEON_AUTH_COOKIE_SECRET (32+ chars); Better Auth 1.4.18 validates HMAC before returning session |
| T-41-02 | Tampering | Cookie session payload | mitigate | HMAC signature check inside Neon Auth; any modification invalidates the session and getSession returns null |
| T-41-03 | Elevation of Privilege | Whitelist bypass | mitigate | Case-insensitive email comparison; whitelist check runs AFTER session resolves, so no way to pass session without email |
| T-41-04 | Information Disclosure | 401 error messages | mitigate | Generic "No active session" vs specific "not authorized" — intentional: attacker can already tell whitelist exists via sign-up failure; no additional leakage |
| T-41-05 | Denial of Service | /api/auth/[...path] open to unauthenticated traffic | accept | Single-user app, low-value target, Vercel platform rate limits apply. Neon Auth also has built-in brute-force protection on sign-in |
| T-41-06 | Repudiation | Sign-in events | transfer | Neon Auth logs sign-in attempts server-side (managed by Neon). No app-level audit logging required in this plan |
</threat_model>
