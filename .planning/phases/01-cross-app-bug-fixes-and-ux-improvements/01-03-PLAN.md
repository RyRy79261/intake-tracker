  : undefined;

After this existing check, add a time-level filter for the creation day:

```typescript
// D-03: On creation day, skip schedule times earlier than prescription creation time
if (createdDate && dateStr === createdDate && prescription.createdAt) {
  const createdTime = new Date(prescription.createdAt);
  const createdMinutes = createdTime.getHours() * 60 + createdTime.getMinutes();
  const scheduleMinutes = (() => {
    const localTime = formatLocalTime(schedule.scheduleTimeUTC, tz);
    const parts = localTime.split(":").map(Number);
    return (parts[0] ?? 0) * 60 + (parts[1] ?? 0);
  })();
  if (scheduleMinutes < createdMinutes) continue;
}
```

This ensures that on the day a prescription is created, dose slots for times before the creation time are not shown. The `formatLocalTime` function is already available in the same file.
