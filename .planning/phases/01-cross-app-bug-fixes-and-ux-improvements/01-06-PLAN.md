In `src/lib/analytics-service.ts`, find the `adherenceRate` function (around line 258). The bug is on the line:

```typescript
const dayTotal = filteredSlots.length;
```

This counts ALL slots including future pending ones for today. For past dates, all slots should count (they're either taken/skipped/missed). For today, only count slots that are in the past OR already actioned (taken/skipped).

Replace the forEach body with time-aware filtering. Reuse the same pattern from `computeProgress()` in medication-ui-utils.ts:

```typescript
const todayStr = new Date().toISOString().split("T")[0];
const now = new Date();
const nowMinutes = now.getHours() * 60 + now.getMinutes();

scheduleMap.forEach((slots, date) => {
  const filteredSlots = prescriptionId
    ? slots.filter((s) => s.prescriptionId === prescriptionId)
    : slots;

  // D-16: For today, exclude future pending doses from the denominator
  const countableSlots = date === todayStr
    ? filteredSlots.filter((s) => {
        const parts = s.localTime.split(":").map(Number);
        const slotMinutes = (parts[0] ?? 0) * 60 + (parts[1] ?? 0);
        const isFuture = slotMinutes > nowMinutes;
        // Count past slots + any future slots already actioned
        return !isFuture || s.status === "taken" || s.status === "skipped";
      })
    : filteredSlots;

  const dayTotal = countableSlots.length;
  const dayTaken = countableSlots.filter((s) => s.status === "taken").length;

  totalSlots += dayTotal;
  totalTaken += dayTaken;

  if (dayTotal > 0) {
    dailyEntries.push({
      date,
      rate: dayTaken / dayTotal,
      taken: dayTaken,
      total: dayTotal,
    });
  }
});
```

The `todayStr` must use the same date format as the scheduleMap keys (YYYY-MM-DD).

In `src/hooks/use-analytics-queries.ts`, the `useInsights` function currently has hardcoded thresholds:
- Line 231: `adherenceData.value.rate < 0.8` (adherence trigger)

Since `useInsights` uses `useLiveQuery` (an async callback), it cannot call React hooks inside the callback. Instead, read the thresholds BEFORE the `useLiveQuery` call:

```typescript
export function useInsights(range: TimeRange) {
  // Read thresholds from settings store (outside useLiveQuery callback)
  const insightThresholds = useSettingsStore((s) => s.insightThresholds);
  const adherenceThreshold = (insightThresholds.adherence_drop ?? 80) / 100; // stored as %, used as decimal
  const fluidDeficitThreshold = (insightThresholds.fluid_deficit ?? 50) / 100; // stored as %, used as decimal

  return useLiveQuery(
    async () => {
      // ... existing code ...

      // Replace hardcoded 0.8 with adherenceThreshold:
      if (adherenceData.value.rate < adherenceThreshold) {
        insights.push({
          // ...
          description: `Adherence is ${Math.round(adherenceData.value.rate * 100)}% (target: ${Math.round(adherenceThreshold * 100)}%).`,
          threshold: adherenceThreshold,
          // ...
        });
      }

      // Replace hardcoded 0.5 with fluidDeficitThreshold:
      if (deficitRatio > fluidDeficitThreshold) {
        insights.push({
          // ...
          threshold: fluidDeficitThreshold,
          // ...
        });
      }

      // BP trend and weight anomaly thresholds can remain hardcoded
      // (they are statistical, not user-configurable)
    },
    [range.start, range.end, adherenceThreshold, fluidDeficitThreshold], // Add thresholds to deps
    [] as Insight[],
  );
}
```
