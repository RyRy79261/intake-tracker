    **Step 1: Define the form validation pattern**

    Establish a consistent pattern for form validation across all forms:
    ```typescript
    // Co-located at top of component file
    const WeightFormSchema = z.object({
      weight: z.number({ invalid_type_error: "Weight is required" })
        .positive("Weight must be positive")
        .max(1000, "Weight seems too high"),
      note: z.string().max(200, "Note too long").optional(),
    });

    // In the component:
    const [fieldErrors, setFieldErrors] = useState<Record<string, string>>({});

    const handleSubmit = () => {
      const parsed = WeightFormSchema.safeParse({ weight, note });
      if (!parsed.success) {
        const errors: Record<string, string> = {};
        for (const issue of parsed.error.issues) {
          const field = issue.path[0];
          if (field && typeof field === "string") errors[field] = issue.message;
        }
        setFieldErrors(errors);
        return;
      }
      setFieldErrors({});
      // proceed with mutation using parsed.data
    };

    // In JSX, under each field:
    {fieldErrors.weight && (
      <p className="text-sm text-destructive mt-1">{fieldErrors.weight}</p>
    )}
    ```
