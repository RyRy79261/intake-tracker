    ```typescript
    import { test, expect } from "@playwright/test";
    import { randomUUID } from "node:crypto";

    type DevIntake = { add: (args: { type: string; amount: number; timestamp: number }) => Promise<unknown> };
    type SyncDev = { pushNow: () => Promise<void>; pullNow: () => Promise<void>; getQueueDepth: () => Promise<number> };

    test.describe("sync-engine", () => {
      test("push — 50 intake records flush to Neon branch", async ({ page }) => {
        await page.goto("/");
        await page.waitForFunction(() => !!(window as unknown as { __devIntake?: DevIntake }).__devIntake);

        for (let i = 0; i < 50; i++) {
          await page.evaluate(async (idx) => {
            const w = window as unknown as { __devIntake: DevIntake };
            await w.__devIntake.add({ type: "water", amount: 250, timestamp: Date.now() - idx * 1000 });
          }, i);
        }

        await page.evaluate(() => (window as unknown as { __syncEngine: SyncDev }).__syncEngine.pushNow());
        await expect.poll(async () => {
          const res = await page.request.get("/api/__test__/count-intake");
          return (await res.json()).count;
        }, { timeout: 15_000 }).toBe(50);
      });

      test("pull — server-written row arrives in Dexie after pullNow", async ({ page }) => {
        await page.goto("/");
        await page.waitForFunction(() => !!(window as unknown as { __syncEngine?: SyncDev }).__syncEngine);
        const id = randomUUID();
        const timestamp = Date.now();

        await page.request.post("/api/__test__/insert-intake", {
          data: { id, type: "water", amount: 500, timestamp, createdAt: timestamp, deletedAt: null },
        });

        await page.evaluate(() => (window as unknown as { __syncEngine: SyncDev }).__syncEngine.pullNow());

        await expect.poll(async () => {
          return await page.evaluate(async (rowId) => {
            // db is already a live singleton in the app bundle — read via a small helper exposed
            // on window.__devIntake (extend Plan 06 helper to include `get(id)` if preferred),
            // OR use the existing Dexie instance via an inline attach. Safest: expose
            // `window.__devIntake.getIntake(id)` alongside .add() in Part B.
            const w = window as unknown as { __devIntake: DevIntake & { getIntake: (id: string) => Promise<{ amount: number } | undefined> } };
            const row = await w.__devIntake.getIntake(rowId);
            return row?.amount ?? null;
          }, id);
        }, { timeout: 10_000 }).toBe(500);
      });

      test("offline reconnect — queued writes flush after setOffline(false)", async ({ page, context }) => {
        await page.goto("/");
        await page.waitForFunction(() => !!(window as unknown as { __devIntake?: DevIntake }).__devIntake);

        for (let i = 0; i < 5; i++) {
          await page.evaluate(async (idx) => {
            const w = window as unknown as { __devIntake: DevIntake };
            await w.__devIntake.add({ type: "water", amount: 100, timestamp: Date.now() - idx * 1000 });
          }, i);
        }
        await page.evaluate(() => (window as unknown as { __syncEngine: SyncDev }).__syncEngine.pushNow());

        await context.setOffline(true);
        for (let i = 5; i < 10; i++) {
          await page.evaluate(async (idx) => {
            const w = window as unknown as { __devIntake: DevIntake };
            await w.__devIntake.add({ type: "water", amount: 100, timestamp: Date.now() - idx * 1000 });
          }, i);
        }

        const depth = await page.evaluate(() => (window as unknown as { __syncEngine: SyncDev }).__syncEngine.getQueueDepth());
        expect(depth).toBe(5);

        await context.setOffline(false);
        await expect.poll(async () => {
          const res = await page.request.get("/api/__test__/count-intake");
          return (await res.json()).count;
        }, { timeout: 30_000 }).toBe(10);
      });
    });
    ```
