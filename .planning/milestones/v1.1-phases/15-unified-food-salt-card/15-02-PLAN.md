                </div>
                <span className="font-semibold text-lg uppercase tracking-wide">
                  Food + Salt
                </span>
              </div>
              {latestEating ? (
                <p className="text-xs text-muted-foreground">
                  {formatDateTime(latestEating.timestamp)}
                </p>
              ) : null}
            </div>

            {/* Food section */}
            <FoodSection />
