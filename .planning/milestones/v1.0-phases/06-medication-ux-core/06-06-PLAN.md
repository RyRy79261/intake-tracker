<objective>
Build the Prescriptions tab -- a list of active prescriptions with detail drawer for viewing/editing schedule, managing titration phases, and adding notes.

Purpose: Users need a dedicated place to manage the "doctor's order" side of their medications -- seeing current dosage, planning titration changes, and keeping prescription-level notes. This separates prescription management (compound, dose, schedule) from inventory management (brand, stock, refills) which stays on the Medications tab.

Output: Prescriptions tab with list view, detail drawer with schedule/phases/notes, titration phase creation and activation.
</objective>

export interface Prescription {
  id: string;

  indication: string;
  notes?: string;
  contraindications?: string[];
  warnings?: string[];
  isActive: boolean;
  createdAt: number;
  updatedAt: number;
  deletedAt: number | null;
  deviceId: string;
}

export interface MedicationPhase {
  id: string;
  prescriptionId: string;
  type: "maintenance" | "titration";
  status: "active" | "completed" | "cancelled" | "pending";
  unit: string;
  foodInstruction: FoodInstruction;
  foodNote?: string;
  startDate?: number;
  endDate?: number;
  createdAt: number;
  updatedAt: number;
  deletedAt: number | null;
  deviceId: string;
}

export interface PhaseSchedule {
  id: string;
  phaseId: string;
  time: string;           // HH:MM local format
  scheduleTimeUTC: number; // minutes since midnight UTC
  dosage: number;
  daysOfWeek: number[];
  enabled: boolean;
  createdAt: number;
  updatedAt: number;
  deletedAt: number | null;
  deviceId: string;
  timezone: string;
}
